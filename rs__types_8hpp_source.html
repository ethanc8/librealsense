161 inline void error::handle(rs2_error* e)
162 {
163 if (e)
164 {
165 auto h = rs2_get_librealsense_exception_type(e);
166 switch (h) {
167 case RS2_EXCEPTION_TYPE_CAMERA_DISCONNECTED:
168 throw camera_disconnected_error(e);
169 case RS2_EXCEPTION_TYPE_BACKEND:
170 throw backend_error(e);
171 case RS2_EXCEPTION_TYPE_INVALID_VALUE:
172 throw invalid_value_error(e);
173 case RS2_EXCEPTION_TYPE_WRONG_API_CALL_SEQUENCE:
174 throw wrong_api_call_sequence_error(e);
175 case RS2_EXCEPTION_TYPE_NOT_IMPLEMENTED:
176 throw not_implemented_error(e);
177 case RS2_EXCEPTION_TYPE_DEVICE_IN_RECOVERY_MODE:
178 throw device_in_recovery_mode_error(e);
179 default:
180 throw error(e);
181 }
182 }
183 }
